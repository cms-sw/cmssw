process.myTrackerAlignment = CalibTracker.Configuration.Common.PoolDBESSource_cfi.poolDBESSource.clone(
    connect = cms.string (databaseFileName),
    toGet = cms.VPSet(
        cms.PSet(

            tag = cms.string('Alignments_0')

    )

        )
